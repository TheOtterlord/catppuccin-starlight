---
title: Install Theme
description: Install the Catppuccin theme for Starlight.
---

import { Steps } from '@astrojs/starlight/components';
import { PackageManagers } from 'starlight-package-managers'

<Steps>

1. Install the theme package to your project with your preferred package manager:
   <PackageManagers pkg="@catppuccin/starlight"/>

2. Add the theme to your Starlight config.
   ```ts title="astro.config.mjs" ins={1,9}
   import catppuccin from "@catppuccin/starlight";

   export default defineConfig({
     // ...
     integrations: [
       starlight({
         // ...
         plugins: [
           catppuccin()
         ]
       })
     ]
   })
   ```

2. (Optional) Customize your theme for light/dark mode using `flavor` and `accent`.
   ```ts title="astro.config.mjs" ins={10-11}
   import catppuccin from "@catppuccin/starlight";

   export default defineConfig({
     // ...
     integrations: [
       starlight({
         // ...
         plugins: [
           catppuccin({ 
             dark: { flavor: "macchiato", accent: "sky" },
             light: { flavor: "latte", accent: "sky" }
           })
          ]
       })
     ]
   })
   ```

</Steps>
